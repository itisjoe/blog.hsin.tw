所以自己加個prototype吧

if (!Array.prototype.indexOf)
{
  Array.prototype.indexOf = function(elt /*, from*/)
  {
    var len = this.length >>> 0;

    var from = Number(arguments[1]) || 0;
    from = (from < 0)
         ? Math.ceil(from)
         : Math.floor(from);
    if (from < 0)
      from += len;

    for (; from < len; from++)
    {
      if (from in this &&
          this[from] === elt)
        return from;
    }
    return -1;
  };
}

[NOTE]
IE8使用 for 時 也會撈出這個indexOf的function 請注意一下

// items 是個array或object
for(var k in items) {
    // IE8會出現該function 請注意一下
    alert(items[k]);
}
